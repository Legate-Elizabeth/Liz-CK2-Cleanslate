trans_saharan_trade_post_trigger = {
	custom_tooltip = {
		text = trans_saharan_province_desc

		OR = {
			province = c_tripolitana	# Trade post
			province = c_fes			# Trade post
			province = c_marrakech		# Trade post
			province = c_timbuktu		# Cloth production
			province = c_aoudaghost		# Salt mine
			province = c_taghaza		# Salt mine
			province = c_bambuk			# Gold mine
			province = c_djimi			# Trade post
			province = c_kano			# Trade post
			province = c_bure			# Gold mine
		}
	}
}
